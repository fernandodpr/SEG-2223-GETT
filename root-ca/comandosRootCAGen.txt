 openssl req -new \
    -config root-ca.conf \
    -out root-ca.csr \
    -keyout private/root-ca.key

openssl ca -selfsign \
    -config root-ca.conf \
    -in root-ca.csr \
    -out root-ca.crt \
    -extensions ca_ext


$ openssl ca \
    -config root-ca.conf \
    -in server.csr \
    -out server.crt \
    -extensions server_ext
